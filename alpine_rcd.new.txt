#!/sbin/openrc-run

# Single reusable app identifier (change this one var/line for new apps)
APP_NAME="pl1"

name="PHP built-in server - ${APP_NAME}"

# Paths (consistent lowercase)
HOME_DIR="/home/${APP_NAME}"
PUBLIC_DIR="${HOME_DIR}/Public"
PID_FILE="/run/${APP_NAME}-server.pid"
ACCESS_LOG="/var/log/${APP_NAME}-access.log"
ERROR_LOG="/var/log/${APP_NAME}-error.log"

command="/usr/bin/php84"
command_args="-d error_log=${ERROR_LOG} -d log_errors=1 -d display_errors=0 -d error_reporting=E_ALL -d cli_server.log_requests=0 -S 0.0.0.0:80 ${PUBLIC_DIR}/index.php -t ${PUBLIC_DIR}/"
command_env="PHP_CLI_SERVER_WORKERS=0"
pidfile="${PID_FILE}"
directory="${HOME_DIR}"

output_log="${ACCESS_LOG}"
error_log="${ERROR_LOG}"

depend() {
    need net
    after network
}

start_pre() {
    checkpath --directory --owner root:root --mode 0755 /var/log

    touch "${output_log}"
    chown root:root "${output_log}"
    chmod 0644 "${output_log}"

    touch "${error_log}"
    chown root:root "${error_log}"
    chmod 0644 "${error_log}"
}

start() {
    ebegin "Starting PHP built-in server"
    cd "${directory}" || { eend 1; return 1; }

    start-stop-daemon --start \
        --pidfile "${pidfile}" \
        --make-pidfile \
        --background \
        --stdout "${output_log}" \
        --stderr "${error_log}" \
        --exec "${command}" -- ${command_args}

    eend $?
}

stop() {
    ebegin "Stopping PHP built-in server"
    start-stop-daemon --stop --pidfile "${pidfile}"
    eend $?
}
