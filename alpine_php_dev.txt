#!/sbin/openrc-run

# Single reusable app identifier (change this one var for new apps)
APP_NAME="ar3"

name="PHP built-in server - ${APP_NAME^}"

# app-specific paths using APP_NAME (auto-capitalized)
AR3_HOME="/home/${APP_NAME}"
AR3_PUBLIC="${AR3_HOME}/Public"
AR3_PID="/run/${APP_NAME}-server.pid"
AR3_ACCESS_LOG="/var/log/${APP_NAME}-access.log"
AR3_ERROR_LOG="/var/log/${APP_NAME}-error.log"

command="/usr/bin/php83"
command_args="-d error_log=${AR3_ERROR_LOG} -d log_errors=1 -d display_errors=0 -d error_reporting=E_ALL -d cli_server.log_requests=0 -S 0.0.0.0:8080 ${AR3_PUBLIC}/index.php -t ${AR3_PUBLIC}/"
command_env="PHP_CLI_SERVER_WORKERS=0"
pidfile="${AR3_PID}"
directory="${AR3_HOME}"

output_log="${AR3_ACCESS_LOG}"
error_log="${AR3_ERROR_LOG}"

depend() {
    need net
    after network
}

start_pre() {
    checkpath --directory --owner root:root --mode 0755 /var/log

    # Access log (stdout)
    touch "${output_log}"
    chown root:root "${output_log}"
    chmod 0644 "${output_log}"

    # Error log (stderr + PHP internal)
    touch "${error_log}"
    chown root:root "${error_log}"
    chmod 0644 "${error_log}"
}

start() {
    ebegin "Starting PHP built-in server"
    cd "${directory}" || { eend 1; return 1; }

    start-stop-daemon --start \
        --pidfile "${pidfile}" \
        --make-pidfile \
        --background \
        --stdout "${output_log}" \
        --stderr "${error_log}" \
        --exec "${command}" -- ${command_args}

    eend $?
}

stop() {
    ebegin "Stopping PHP built-in server"
    start-stop-daemon --stop --pidfile "${pidfile}"
    eend $?
}
